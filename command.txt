


FROM node:6.14.2
EXPOSE 8080
COPY server.js .
CMD node server.js


# Build Image
docker build --tag=nodeimage .

# tag image mcclayac/get-started:part2
$ docker tag nodeimage mcclayac/nodeimage:part2
$ docker tag nodeimage localhost/nodeimage:part2
